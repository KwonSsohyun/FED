[ AJAX (Asynchronous Javascript And XML) ] 
  AJAX란, JavaScript의 라이브러리중 하나이며, 
  Asynchronous Javascript And Xml(비동기식 자바스크립트와 xml)의 약자이다. 
  
  브라우저가 가지고있는 XMLHttpRequest 객체를 이용해서 
  전체 페이지를 새로 고치지 않고도 
  페이지의 일부만을 위한 데이터를 로드하는 기법 이며 
  JavaScript를 사용한 비동기 통신, 클라이언트와 서버간에 XML 데이터를 주고받는 기술이다.

  즉, 쉽게 말하자면 자바스크립트를 통해서 서버에 데이터를 요청하는 것이다.

  ★ HTML, JavaScript 에서 작업한 것들과 ⇔ PHP 가 
     주거니 받거니 통신을 하기 위한것.
     일부통신(비동기)
     XML로 데이터를 주거니받거니 통신

   ■ 예제 파일에서 파일확장자명을 다 .php 로 맞췄다.
     ▷ FED\009.Ajax학습\vogue 
        가서보면 .html 형식이 아닌 .php 형식인것을 볼 수 있다.
     ▶ PHP 언어를 썼기 때문에 호출할때 확장자를 통일해야 하기 때문에 맞춘것
        꼭 PHP 형식이 아니어도 된다. (jsp, asp 도 있다.)

[ Ajax를 이용한 POST 방식으로 DB에 데이터 입력하기! ]
  AJAX = Asynchronous Javascript and XML의 약자

  - 비동기통신이란? 
    : 쉽게말해서 페이지가 새로고쳐지지 않고, 
      그대로 있으면서 일부분만 서버통신을 하는 것을 말한다!
                
  - 제이쿼리는 POST방식으로 ajax를 할 수 있다!


    [ POST 방식 Ajax 메서드 ]
      $.post(URL,data,callback)
      ▶ $.post(전송할 페이지, 전송할 데이터, 전송후 실행함수)  

    ■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■

    ▷ 참고자료 : FED\009.Ajax학습\vogue\js >> "member.js"
    ▶ 회원가입창에서 회원가입한 정보들을 DB에 데이터 넣는 방법

    $.post(
        // 1. 전송할 페이지 
        "process/ins.php",
        
        
        // 2. 전송할 데이터 → {속성:값} ▶ 객체형식
        {
            // 1.아이디
            "mid":$("#mid").val(),
            // 2.비번
            "mpw":$("#mpw").val(),
            // 3.이름
            "mnm":$("#mnm").val(),

            // 4.성별
            "gen":$(":radio[name=gen]:checked").val(),

            // 5-1.이메일 앞주소
            "email1":$("#email1").val(),
            // 5-2.이메일 뒷주소
            "seleml":$("#seleml").val(),
            // 5-3.직접입력 이메일 뒷주소
            "email2":$("#email2").val()

        },

        
        // 3. 전송후 실행함수
        function(res){ 

            /// 성공시! ///
            if(res==="ok"){
                // 메시지 띄우기
                alert("회원가입을 축하드립니다! 짝짝짝!");

                // 로그인페이지로 이동하기 (보안이동)
                location.replace("login.php");

            } //// if문 : 성공시 ///

            else {

                // 실패메시지 찍기
                alert(res);

            } //// else문 : 실패시 ///


        } //////// 전송후 실행함수 /////////////////////


    ); /////// post 메서드 ///////////////////////////////////////////


    ■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■


    ▷ 참고자료 : FED\009.Ajax학습\vogue\js >> "login.js" 
    ▶ 로그인창에서 아이디와 비번을 모두 입력하고 로그인버튼 눌렀을때 DB에 회원정보 조회! (있는 계정인지 확인)

    $.post(

        // 1. 전송할 페이지
        "process/loginSet.php",

        // 2. 전송할 데이터
        {
            "mid" : mid.val(), //아이디
            "mpw" : mpw.val()  // 비번
        },
                
        // 3. 콜백함수
        function(res){ // res - 결과값 전달변수

            // 3-1. 로그인 성공시
            if(res==="ok"){

                // 메인페이지 이동하기
                location.href = "index.php";

            }///// if ////

            // 3-2. 아이디가 없는 경우
            else if(res==="no"){
                alert("사용가능한 ID가 아닙니다!");

                // 아이디, 비번 지우고, 아이디에 포커스 (꼭 필요한건 아니지만 서비스!)
                mid.val("").focus();
                mpw.val("");

            }///// else if ///

            // 3-3. 비밀번호 불일치
            else if(res==="again"){
                alert("비밀번호가 일치하지 않습니다.");

                // 비번지우고 비번에 포커스 (꼭 필요한건 아니지만 서비스!)
                mpw.val("").focus();

            }///// else if ////

        } /////// 콜백함수 ////////////

    ); ////////// post 메서드 ///////////////////////////////

[ AJAX로 중복아이디 검사하기! ]
  1) POST 방식 처리 메서드
     $.post(URL,data,callback)

  2) get 방식 처리 메서드
     $.get(URL,callback)

  3) 위 2가지 유형 중 선택처리 메서드
     $.ajax(전송할페이지, 전송방식, 보낼데이터, 전송할데이터타입, 비동기옵션, 성공처리, 실패처리)  

    ■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■■

    ▷ 참고자료 : FED\009.Ajax학습\vogue\js >> "member.js"
    ▶ 회원가입창에서 이미 가입되어있는 아이디인지 아닌지 아이디를 사용자가 넣었을때 검사하는 기능

    $.ajax({ // 객체형식으로 보냄 {} 
                        
        // 1. 전송할페이지
        url: "process/chkID.php",

        // 2. 전송방식
        type: "post",

        // 3. 보낼데이터
        data: {
            "mid":$("#mid").val()
        },

        // 4. 전송할데이터타입
        dataType: "html",

        // 5. 비동기옵션 (통과여부 false >> pass = false; 썼기에 아래 이게 먹는 것.) 
        async: false,
                        
        // 6. 성공처리
        success: function(res){

            if(res==="ok"){ // DB에 없는 아이디
                $("#mid").siblings(".msg")
                .text("훌륭한 아이디네요~!")
                .addClass("on"); 
            } ////// if ////

            else { // 실패한경우 >> DB에 이미 같은 ID가 있는 경우
                $("#mid").siblings(".msg") 
                .text("사용중인 ID입니다!")
                .removeClass("on");

                // 통과여부 false
                pass = false; // 해줘야 async(어싱크) 먹는다.

            } ////// else ////
        }, ////// success 함수 //////
                        
        // 7. 실패처리
        error: function(xhr,status,error){
            alert("연결실행실패"+error);
        } //// error 함수 /////
                        

    }); //////////// ajax 메서드 /////////////
